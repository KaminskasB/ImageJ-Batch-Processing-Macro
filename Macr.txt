function removeBackground(input, output, filename) {
	open(input + filename);
	run("Subtract Background...", "rolling=50 sliding");
	saveAs("Tiff", output + filename);
	close();
}

input = "/home/benediktaskaminskas/PET Scans/"
output = "/home/benediktaskaminskas/Modified PET Scans/"

setBatchMode(true); 
imageList = getFileList(input);
for (i = 0; i<imageList.length; i++)
	removeBackground(input, output, imageList[i]);
setBatchMode(false);
